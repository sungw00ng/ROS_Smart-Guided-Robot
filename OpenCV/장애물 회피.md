```Python
import time
from pop import Pilot, Lidar

# 객체 생성
bot = Pilot.SerBot()
lidar = Lidar.Rplidar()

# LiDAR 초기화
lidar.connect()
lidar.startMotor()

# 속도 설정
bot.setSpeed(80)

try:
    while True:
        # LiDAR 스캔
        vectors = lidar.getVectors()

        # 전방 범위: 330~360도 + 0~30도
        front = [v for v in vectors if (v[0] >= 330 or v[0] <= 30) and v[1] > 0]

        # 장애물 유무 판단
        obstacle = any(v[1] < 500 for v in front)

        if obstacle:
            print("🚧 장애물 감지됨 → 회피 중")

            # 좌/우 거리 비교
            left = [v[1] for v in vectors if 300 <= v[0] <= 345 and v[1] > 0]
            right = [v[1] for v in vectors if 15 <= v[0] <= 60 and v[1] > 0]
            left_avg = sum(left) / len(left) if left else 0
            right_avg = sum(right) / len(right) if right else 0

            if left_avg > right_avg:
                print("↪ 우회전")
                bot.steering = 0.8
            else:
                print("↩ 좌회전")
                bot.steering = -0.8

            bot.forward()
            time.sleep(0.5)
            bot.stop()

        else:
            print("🟢 직진")
            bot.steering = 0
            bot.forward()

        time.sleep(0.1)

except KeyboardInterrupt:
    print("🛑 프로그램 종료")
    bot.stop()
    lidar.stopMotor()
```
